* [animation](https://bezkoder.com/dart-map/)  
 
```dart
class _LoginState extends State<Login> with TickerProviderStateMixin {
  
  late final AnimationController _controller = AnimationController(
    duration: Duration(milliseconds: fadeMillisec),
    vsync: this,
  )..repeat(reverse: false);
  late final Animation<double> _animation = CurvedAnimation(
    parent: _controller,
    curve: Curves.easeIn,
  );
  // Animation : proverb
  int _count = 0;
  var keys = [0, 1, 2]; // animation switcher 사용하기 위해서는 key가 필요

  @override
  void initState() {
    super.initState();
    _controller.forward(from: 0); // animation 한번만 실행
    changeProverb(); 
  }

  @override
  void dispose() {
    // 한번만 animation 실행 후 중지
    _controller.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Consumer<UserProvider>(builder: (context, provider, child) {
      return FadeTransition(
          opacity: _animation,
          child: Padding(
              child: Scaffold(
                body: Center(
                child: Container(
                  child: Column(
                    children: [
                      _appTitle(),
                    ],
                  ),
                ),
              ))));
    });
  }

  // AnimationSwitcher로 2개의 위젯이 FadeTransion으로 바뀐다
  _appTitle() {
    return Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      children: [
        AnimatedSwitcher(
          duration: const Duration(milliseconds: 800),
          transitionBuilder: (Widget child, Animation<double> animation) {
            return FadeTransition(opacity: animation, child: child);
          },
          child: _proverb(_count),
        ),
      ],
    );
  }
  
  // 5초마다 속담 변경 (animationSwitcher), initState()에서 실행
  Future changeProverb() async {
    while (true) {
      await new Future.delayed(const Duration(seconds: 5), () {
        setState(() {
          _count++;
          if (_count > 2) {
            _count = 0;
          }
        });
      });
    }
  }
```
